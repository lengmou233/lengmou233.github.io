@Service
@Log4j2
public class AliYunPanToken {

    @Autowired
    private AliYunPanProperties aliYunPanProperties;

    private CloseableHttpClient httpClient = HttpClientBuilder.create().build();


    /**
     * 刷新token 获取 access_token
     * @return
     */
    @PostConstruct
    public void refreshToken(){
        HttpPost httpPost = new HttpPost(Const.REFRESH_TOKEN_URI);
        httpPost.setHeader(HttpHeaders.CONTENT_TYPE, "application/x-www-form-urlencoded");
        List<BasicNameValuePair> params = new ArrayList<>();
        params.add(new BasicNameValuePair("refresh_token",aliYunPanProperties.getRefreshToken()));
        UrlEncodedFormEntity formEntity = new UrlEncodedFormEntity(params, StandardCharsets.UTF_8);
        httpPost.setEntity(formEntity);
        CloseableHttpResponse response = null;
        try {
            response = httpClient.execute(httpPost);
            HttpEntity entity = response.getEntity();
            String s = EntityUtils.toString(entity);
            JSONObject resJson = JSON.parseObject(s);
            aliYunPanProperties.setRefreshToken(resJson.getString("refresh_token"));
            aliYunPanProperties.setAccessToken(resJson.getString("access_token"));
            aliYunPanProperties.setDeviceId(resJson.getString("default_drive_id"));
            log.info("刷新token成功!!");

        } catch (Exception e) {
            e.printStackTrace();
            log.error("刷新token失败，请检查refresh_token是否过期");
        }
    }


}

/**
* 每两小时执行一次 刷新token 因为access_token 每两小时过期一次
*/
@Scheduled(cron = "0 0 0/2 * * *")
    public void cron() {
        log.info("定时任务：开始刷新token");
        aliYunPanToken.refreshToken();
    }

    private CloseableHttpClient httpClient = HttpClientBuilder.create().build();

    /**
     * 判断有没有过期
     * @return
     */
    public CloseableHttpResponse getHttpResponse(HttpPost httpPost){
        try {
            CloseableHttpResponse response = httpClient.execute(httpPost);
            if(response.getStatusLine().getStatusCode()!=200){
                aliYunPanToken.refreshToken();
                return httpClient.execute(httpPost);
            }
            return response;
        } catch (IOException e) {
            e.printStackTrace();
        }
        return null;
    }

    /**
     * 获取用户基本信息
     */

    public JSONObject getUser(){

        httpPost.setHeader(HttpHeaders.CONTENT_TYPE, "application/json;charset=utf8");
        StringEntity stringEntity = new StringEntity("{}", "utf-8");
        httpPost.setEntity(stringEntity);
        CloseableHttpResponse httpResponse = getHttpResponse(httpPost);
        JSONObject jsonObject = parseResponseToJSON(httpResponse);

    }
